```typescript
// No D3 force simulation - pure static positioning
function calculateRadialPositions(
  player: Empire,
  immediate: Empire[],
  distant: Empire[]
): Map<string, Position> {
  const positions = new Map<string, Position>();
  const center = { x: 450, y: 300 };

  // Player at center (fixed)
  positions.set(player.id, center);

  // Immediate neighbors: inner ring (150px)
  immediate.forEach((empire, i) => {
    const angle = (i / immediate.length) * 2 * Math.PI;
    positions.set(empire.id, {
      x: center.x + Math.cos(angle) * 150,
      y: center.y + Math.sin(angle) * 150,
    });
  });

  // Distant neighbors: outer ring (300px)
  distant.forEach((empire, i) => {
    const angle = (i / distant.length) * 2 * Math.PI;
    positions.set(empire.id, {
      x: center.x + Math.cos(angle) * 300,
      y: center.y + Math.sin(angle) * 300,
    });
  });

  return positions;
}
```

### Concept 2: Sector System Schema

```sql
-- Database additions for regional system
ALTER TABLE empires ADD COLUMN sector_id VARCHAR(50);

CREATE TABLE sectors (
  id VARCHAR(50) PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  position_x INTEGER NOT NULL,
  position_y INTEGER NOT NULL,
  game_id UUID REFERENCES games(id) ON DELETE CASCADE
);

CREATE TABLE sector_connections (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  game_id UUID REFERENCES games(id) ON DELETE CASCADE,
  from_sector_id VARCHAR(50) NOT NULL,
  to_sector_id VARCHAR(50) NOT NULL,
  connection_type VARCHAR(20) NOT NULL, -- 'natural' | 'wormhole'
  attack_cost_modifier DECIMAL DEFAULT 1.0,
  UNIQUE(game_id, from_sector_id, to_sector_id)
);

-- Enforce sector-based attacks
CREATE OR REPLACE FUNCTION validate_attack_sectors()
RETURNS TRIGGER AS $$
BEGIN
  -- Check if attacker and defender are in connected sectors
  IF NOT EXISTS (
    SELECT 1 FROM sector_connections sc
    JOIN empires attacker ON attacker.id = NEW.attacker_id
    JOIN empires defender ON defender.id = NEW.defender_id
    WHERE sc.game_id = NEW.game_id
      AND (
        (sc.from_sector_id = attacker.sector_id AND sc.to_sector_id = defender.sector_id)
        OR
        (attacker.sector_id = defender.sector_id) -- Same sector always OK
      )
  ) THEN
    RAISE EXCEPTION 'Cannot attack: empires not in connected sectors';
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER check_sector_attack
  BEFORE INSERT ON attacks
  FOR EACH ROW EXECUTE FUNCTION validate_attack_sectors();
```

### Concept 3: Filter State Management

```typescript
type FilterMode = 'immediate' | 'expanded' | 'galaxy';

interface StarmapState {
  filterMode: FilterMode;
  empires: {
    immediate: Empire[];
    distant: Empire[];
    all: Empire[];
  };
  selectedEmpire: string | null;
}

function getVisibleEmpires(mode: FilterMode, state: StarmapState): Empire[] {
  switch (mode) {
    case 'immediate':
      return state.empires.immediate; // 3-5 empires
    case 'expanded':
      return [
        ...state.empires.immediate,
        ...state.empires.distant
      ]; // 15-20 empires
    case 'galaxy':
      return state.empires.all; // All 100 empires
  }
}

function renderMap(mode: FilterMode, empires: Empire[]) {
  if (mode === 'immediate') {
    // Static radial layout (NO D3)
    return <RadialMap empires={empires} />;
  } else if (mode === 'expanded') {
    // Light force-directed (D3 with fewer nodes)
    return <ForceDirectedMap empires={empires} strength={-150} />;
  } else {
    // Full force-directed (D3 optimized for 100 nodes)
    return <ForceDirectedMap empires={empires} strength={-50} simplified />;
  }
}
```
